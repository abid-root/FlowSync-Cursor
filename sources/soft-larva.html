/* effect key: soft-larva */
function initSoftLarva(section) {
  const layer = section.querySelector('.cursor-layer');
  const parts = [];

  function createBody(x, y) {
    for (let i = 0; i < 14; i++) {
      const node = document.createElement('span');
      node.className = 'segment';
      node.style.position = 'absolute';
      node.style.width = Math.max(7, 19 - i * 0.72) + 'px';
      node.style.height = node.style.width;
      node.style.borderRadius = '999px';
      node.style.background = i === 0 ? '#fff7ea' : 'rgba(242,237,228,.86)';
      node.dataset.x = x;
      node.dataset.y = y;
      layer.appendChild(node);
      parts.push(node);
    }
  }

  section.addEventListener('pointermove', (event) => {
    const rect = section.getBoundingClientRect();
    let tx = event.clientX - rect.left;
    let ty = event.clientY - rect.top;

    if (!parts.length) createBody(tx, ty);

    parts.forEach((node, index) => {
      const px = Number(node.dataset.x);
      const py = Number(node.dataset.y);
      const ease = index === 0 ? 0.28 : 0.18;
      const nx = px + (tx - px) * ease;
      const ny = py + (ty - py) * ease;
      node.dataset.x = nx;
      node.dataset.y = ny;
      node.style.transform = `translate(${nx}px, ${ny}px) translate(-50%, -50%)`;
      tx = nx;
      ty = ny;
    });
  });
}

document.querySelectorAll('[data-cursor-effect="soft-larva"]').forEach(initSoftLarva);
