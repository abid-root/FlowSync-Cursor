/* effect key: static-box */
function initStaticBoxes(section) {
  const layer = section.querySelector('.cursor-layer');
  let last = 0;

  function spawn(x, y) {
    const node = document.createElement('span');
    node.className = 'piece';
    node.style.position = 'absolute';
    node.style.left = x + 'px';
    node.style.top = y + 'px';
    node.style.width = '12px';
    node.style.height = '12px';
    node.style.borderRadius = '999px';
    node.style.background = 'var(--accent)';
    node.style.transform = 'translate(-50%, -50%)';
    node.animate([
      { opacity: 0, transform: 'translate(-50%, -50%) scale(.7)' },
      { opacity: .9, offset: .1 },
      { opacity: 0, transform: 'translate(calc(-50% + 36px), calc(-50% - 32px)) scale(1.6)' }
    ], { duration: 1100, easing: 'ease-out' }).onfinish = () => node.remove();
    layer.appendChild(node);
  }

  section.addEventListener('pointermove', (event) => {
    const now = performance.now();
    if (now - last < 96) return;
    last = now;
    const rect = section.getBoundingClientRect();
    spawn(event.clientX - rect.left, event.clientY - rect.top);
  });
}

document.querySelectorAll('[data-cursor-effect="static-box"]').forEach(initStaticBoxes);
