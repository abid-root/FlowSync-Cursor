  <div class="fx-layer" aria-hidden="true"></div>

.cursor-demo {
  position: relative;
  min-height: 320px;
  overflow: hidden;
  border-radius: 24px;
  background: #0f1726;
}

.fx-layer {
  position: absolute;
  inset: 0;
  pointer-events: none;
}

.preview-zone {
  position: absolute;
  inset: 0;
}
const effect = COLD_EFFECTS.find((item) => item.key === "ink-drop");
const target = document.querySelector("[data-cursor-effect='ink-drop']");
const layer = target.querySelector(".fx-layer");
const preview = target.querySelector(".preview-zone") || target;

preview.addEventListener("pointermove", (event) => {
  const rect = layer.getBoundingClientRect();
  COLD_FX.spawn(effect, layer, event.clientX - rect.left, event.clientY - rect.top);
});

preview.addEventListener("pointerleave", () => {
  if (["snake", "centipede", "jelly", "fish", "flowsync-basic"].includes(effect.kind)) {
    COLD_FX.clear(layer);
  }
});
