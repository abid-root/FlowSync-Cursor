Piston Push

A piston shaft pushes forward then retracts.

copy source

Use this effect

<section class="cursor-demo" data-cursor-effect="u80-piston-push">
  <div class="fx-layer" aria-hidden="true"></div>
</section>

<link rel="stylesheet" href="assets/style.css">
<script src="assets/data.js"></script>
<script src="assets/ultra80-data.js"></script>
<script src="assets/fx.js"></script>
<script src="assets/ultra80-cursors.js"></script>
.cursor-demo {
  position: relative;
  min-height: 320px;
  overflow: hidden;
  border-radius: 24px;
}

.fx-layer {
  position: absolute;
  inset: 0;
  pointer-events: none;
}

/* Effect styles are in assets/style.css under:
   ULTRA 80 ADD-ONLY CURSOR PACK */
const effect = COLD_EFFECTS.find((item) => item.key === "u80-piston-push");
const target = document.querySelector("[data-cursor-effect='u80-piston-push']");
const layer = target.querySelector(".fx-layer");

target.addEventListener("pointermove", (event) => {
  const rect = layer.getBoundingClientRect();
  COLD_FX.spawn(effect, layer, event.clientX - rect.left, event.clientY - rect.top);
});
Piston Push is rendered by assets/ultra80-cursors.js.
